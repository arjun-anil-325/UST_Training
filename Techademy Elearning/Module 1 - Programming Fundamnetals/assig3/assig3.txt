START

DECLARE n, i
DECLARE m1, m2, m3
DECLARE total, average
DECLARE grade

PRINT "Enter number of students"
INPUT n

FOR i = 1 TO n

    PRINT "Enter marks of student", i

    INPUT m1
    INPUT m2
    INPUT m3

    IF m1 < 0 OR m1 > 100 OR
       m2 < 0 OR m2 > 100 OR
       m3 < 0 OR m3 > 100 THEN

        PRINT "Invalid marks"

    ELSE
        total = m1 + m2 + m3
        average = total / 3

        IF average >= 90 THEN
            grade = 'A'
        ELSE IF average >= 75 THEN
            grade = 'B'
        ELSE IF average >= 60 THEN
            grade = 'C'
        ELSE IF average >= 40 THEN
            grade = 'D'
        ELSE
            grade = 'F'
        ENDIF

        PRINT average
        PRINT grade
    ENDIF

END FOR

END
